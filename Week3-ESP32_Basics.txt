6. Switching on different LED depending on the TouchPad capacitance drop
  from machine import Pin, TouchPad
  import time
  light1 = Pin(12, Pin.OUT)
  light2 = Pin(18, Pin.OUT)
  touch = TouchPad(Pin(4))

  while True:
      touch_value = touch.read()
      print(touch_value)
      time.sleep(0.1)
      if touch_value<200:
          light1.on()
          light2.off()
      if touch_value>=200:
          light1.off()
          light2.on()
# Output = The LED will switch on depending on the drop in capacitance i.e, if the capcitance < 200, light 1 is on, if the capacitance >= 200, light 2 is on
